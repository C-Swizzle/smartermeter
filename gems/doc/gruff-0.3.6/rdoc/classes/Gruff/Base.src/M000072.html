    def draw_title
      return if (@hide_title || @title.nil?)

      @d.fill = @font_color
      @d.font = @font if @font
      @d.stroke('transparent')
      @d.pointsize = scale_fontsize(@title_font_size)
      @d.font_weight = BoldWeight
      @d.gravity = NorthGravity
      @d = @d.annotate_scaled( @base_image,
                               @raw_columns, 1.0,
                               0, @top_margin,
                               @title, @scale)
    end
