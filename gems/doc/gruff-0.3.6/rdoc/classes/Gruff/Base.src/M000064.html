    def setup_drawing
      # Maybe should be done in one of the following functions for more granularity.
      unless @has_data
        draw_no_data()
        return
      end

      normalize()
      setup_graph_measurements()
      sort_norm_data() if @sort # Sort norm_data with avg largest values set first (for display)

      draw_legend()
      draw_line_markers()
      draw_axis_labels()
      draw_title
    end
