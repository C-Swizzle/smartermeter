    def draw
      make_stacked if @stacked
      setup_drawing

      debug {
        # Outer margin
        @d.rectangle( @left_margin, @top_margin,
                      @raw_columns - @right_margin, @raw_rows - @bottom_margin)

        @d.rectangle( @graph_left, @graph_top, @graph_right, @graph_bottom)
      }
    end
