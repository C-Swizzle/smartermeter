    def draw_no_data
      @d.fill = @font_color
      @d.font = @font if @font
      @d.stroke('transparent')
      @d.font_weight = NormalWeight
      @d.pointsize = scale_fontsize(80)
      @d.gravity = CenterGravity
      @d = @d.annotate_scaled( @base_image,
                               @raw_columns, @raw_rows/2.0,
                               0, 10,
                               @no_data_message, @scale)
    end
