  def init_photo_bar_graphics    
    color_list = Array.new
    theme_dir = File.dirname(__FILE__) + '/../../assets/' + theme

    Dir.open(theme_dir).each do |file|
      next unless /\.png$/.match(file)
      color_list << Image.read("#{theme_dir}/#{file}").first
    end
    @colors = color_list
  end
