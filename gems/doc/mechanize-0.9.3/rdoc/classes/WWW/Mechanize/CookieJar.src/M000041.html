      def dump_cookiestxt(io)
        to_a.each do |cookie|
          fields = []
          fields[0] = cookie.domain

          if cookie.domain =~ /^\./
            fields[1] = "TRUE"
          else
            fields[1] = "FALSE"
          end

          fields[2] = cookie.path

          if cookie.secure == true
            fields[3] = "TRUE"
          else
            fields[3] = "FALSE"
          end

          fields[4] = cookie.expires.to_i.to_s

          fields[5] = cookie.name
          fields[6] = cookie.value
          io.puts(fields.join("\t"))
        end
      end
