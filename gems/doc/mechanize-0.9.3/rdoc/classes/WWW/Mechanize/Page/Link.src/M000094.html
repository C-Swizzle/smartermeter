        def initialize(node, mech, page)
          @node = node
          @href = node['href'] 
          @text = node.inner_text
          @page = page
          @mech = mech
          @attributes = node

          # If there is no text, try to find an image and use it's alt text
          if (@text.nil? || @text.length == 0) && node.search('img').length > 0
            @text = ''
            node.search('img').each do |e|
              @text << ( e['alt'] || '')
            end
          end

        end
