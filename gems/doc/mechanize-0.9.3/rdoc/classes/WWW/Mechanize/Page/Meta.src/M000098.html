          def parse(content, uri)
            if content && content =~ CONTENT_REGEXP
              delay, url = $1, $3

              url = case url
              when nil, "" then uri.to_s
              when /^http/i then url
              else "http://#{uri.host}#{url}"
              end

              block_given? ? yield(delay, url) : [delay, url]
            else
              nil
            end
          end
