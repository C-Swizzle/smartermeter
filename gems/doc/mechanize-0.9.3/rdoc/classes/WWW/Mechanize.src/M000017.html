    def head(url, query_params = {}, options = {})
      options = {
        :uri      => url,
        :headers  => {},
        :params   => query_params,
        :verb     => :head
      }.merge(options)
      # fetch the page
      page = fetch_page(options)
      yield page if block_given?
      page
    end
